═══ PLATFORM DETECTION (use in app code) ═══

Detect platform and load the correct asset:

const platform = (() => {
  const ua = navigator.userAgent;
  if (/android/i.test(ua)) return 'android';
  if (/iPad|iPhone|iPod/.test(ua) && !window.MSStream) return 'ios';
  if (window.matchMedia('(display-mode: standalone)').matches) return 'pwa';
  return 'web';
})();

const splashAsset = {
  android: '/assets/splashAndroid.png',
  ios:     '/assets/splashIOS.png',
  pwa:     '/assets/splashWeb.png',
  web:     '/assets/splashWeb.png',
}[platform];

// Show splash on load, hide after 2.5s
const splash = document.getElementById('splash');
if (splash) {
  splash.style.backgroundImage = `url(${splashAsset})`;
  setTimeout(() => splash.classList.add('hidden'), 2500);
}
